	(slurp "beja-atmaan-pdgms-trial.edn")))
(def lang (name (pdgmmap :lang)))
(def Lang (clojure.string/capitalize lang))
(def sgpref (pdgmmap :sgpref))
(def x ( map println [
		(format "\n#SCHEMATA:\n")
		(format "aama:%s a aamas:Language ." Lang)

	])
